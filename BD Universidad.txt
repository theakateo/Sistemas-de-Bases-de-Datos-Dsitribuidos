 CREATE TABLE CAMPUS
(ID_CAM VARCHAR(5) PRIMARY KEY,
NOM_CAM VARCHAR(20) NOT NULL,
UBI_CAM VARCHAR(20) NOT NULL);

CREATE TABLE CARRERAS_HUACHI
(ID_CAR VARCHAR(5) PRIMARY KEY,
NOM_CAR VARCHAR(50) NOT NULL,
CAM_PER_CAR REFERENCES CAMPUS(ID_CAM),
CUPOS NUMBER NOT NULL);

CREATE TABLE CARRERAS_INGAHURCO
(ID_CAR VARCHAR(5) PRIMARY KEY,
NOM_CAR VARCHAR(50) NOT NULL,
CAM_PER_CAR REFERENCES CAMPUS(ID_CAM),
CUPOS NUMBER NOT NULL);

CREATE TABLE CARRERAS_QUEROCHACA
(ID_CAR VARCHAR(5) PRIMARY KEY,
NOM_CAR VARCHAR(50) NOT NULL,
CAM_PER_CAR REFERENCES CAMPUS(ID_CAM),
CUPOS NUMBER NOT NULL);
